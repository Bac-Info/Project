Definition :

Un algorithme d'approximation permet de chercher une valeur approchée pour des valeurs qui ne sont pas des valeurs exactes.

Recherche d'un point fixe d une fonction f : 
program point_fixe ;
uses wincrt ;
Const eps = 1e-10 ;
var x,xp : real ; i : integer;
   Function f ( x : real ):real ;
   begin
       f := sqrt(x+1);
   end;
begin
x:= 1 ;  i:=0;
Repeat
    xp:= x ;
    x:= f(x);
    i:= i +1;
until abs( x-xp )< eps ;
writeln('le point fixe de cette fonction est ',x,'trouvé après ',i,'iteration');
End.

function rectangle (a,b :reel ; n : integer ) : real ;
var  s, h, x : real;
i: integer;
begin
   s:= 0; h:= (b- a) /n ;  x:= a;
   for i := 1 to n do
    begin
      s:= s + h * f(x) ;
      x:= x + h ;
    end;
   rectangle:= s ;
End;
Méthode des trapèzes : 
function trapezes (a,b :reel ; n : integer ) : real ;
var  s, h, x : real;
i: integer;
begin
   s:= 0; h:= (b- a) /n ;  x:= a;
   for i := 1 to n do
    begin
      s:= s + (h/2) * (f(x)+ f(x+h) );
      x:= x + h ;
    end;
   trapezes:= s ;
End;

Function leonard( eps : real ): real;
var n: integer;
s,sp:real;
begin
   n:= 1;  s:= 1;

Function Leibniz ( eps :real ) : real;
Var   j,signe : integer ;
terme , s: real;
begin
     terme := 1 ; j:= -1 ;
     signe := -1;  s:= 0 ;
  while ( terme >= eps ) do
   begin
      j:= j+2;
