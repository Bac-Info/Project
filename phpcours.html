Fonctions utiles pour les variables

gettype($nom_var);

				
Rôle: 
Détèrminer le type de données de la variable.

				
Exemple: 
Pour $x = 2.3 la résultat sera: Double

			
settype($nom_var,'type');

				
Rôle: 
Définit explicitement le type de variable.

				
Exemple: 
Pour $x = 2.3 et 'type'= 'Integer' la résultat sera: $x = 2

			
isset($nom_var);

				
Rôle: 
Voir si une variable possède une valeur ou non.

				
Exemple: 
Pour $x = 2.3 la résultat sera: true 

			
unset($nom_var);

				
Rôle: 
détruit une variable

				
Exemple: 
Pour unset($x) la résultat de isset($x) sera: false

			
empty($nom_var);

				
Rôle: 
Renvoie true si la variable est vide ou vaut 0, sinon renvoie false

				
Exemple: 
Pour $x = 2.3 la résultat sera: false 

			
is_'type'($nom_var);

				
Rôle: 
Détermine si la variable est un 'type'

				
Exemple: 
Pour $x = 2.3 la résultat de is_double($x) sera: True

			
intval($nom_var);

				
Rôle: 
Définit un entier.

				
Exemple: 
Pour $x = '2.3 degrés' la résultat sera: 2

			
doubleval($nom_var);

				
Rôle: 
Définit un double.

				
Exemple: 
Pour $x = '2.3 degrés' la résultat sera: 2.3

			
strval($nom_var);

				
Rôle: 
Définit une chaîne.

				
Exemple: 
Pour $x = '2.3 degrés' la résultat sera: 2.3 degrés

Fonctions utiles pour les chaînes

chr('code ASCII');

				
Rôle: 
Retourne le caractère dont le code ASCII est donnée.

				
Exemple: 
Pour 'code ASCII' = 65 la résultat sera: 'A'

			
ord($ch);

				
Rôle: 
Retourne le code ASCII du premier caractère d'une chaîne.

				
Exemple: 
Pour $ch = 'A' la résultat sera: 65

			
strlen($ch);

				
Rôle: 
Retourne la longueur d'une chaîne de caractère.

				
Exemple: 
Pour $ch = 'Velzimar' la résultat sera: 8

			
strtoupper($ch);

				
Rôle: 
Mettre une chaîne de caractère en majuscule.

				
Exemple: 
Pour $ch = 'Velzimar' la résultat sera: 'VELZIMAR'

			
strtolower($ch);

				
Rôle: 
Mettre une chaîne de caractère en minuscule.

				
Exemple: 
Pour $ch = 'Velzimar' la résultat sera: 'velzimar'

			
strcmp($ch1,$ch2);

				
Rôle: 
Renvoie la valeur 0 si les deux chaînes sont identiques
(On tient compte sur les majuscules et les minuscules).

				
Exemple: 
Pour $ch1 = 'Velzimar' et $ch2 = 'VELZIMAR' la résultat sera: <> 0

			
strcasecmp($ch1,$ch2);

				
Rôle: 
Renvoie la valeur 0 si les deux chaînes sont identiques
(On ne tient pas compte sur les majuscules et les minuscules).

				
Exemple: 
Pour $ch1 = 'Velzimar' et $ch2 = 'VELZIMAR' la résultat sera: 0

			
trim($ch);

				
Rôle: 
Effacer les espaces de début et de fin de chaîne.

				
Exemple: 
Pour $ch = '   Velzimar   ' la résultat sera: 'Velzimar'

			
substr($ch,$pos,$long);

				
Rôle: 
Extraire une sous-chaîne de caractère.

				
Exemple: 
Pour $ch = 'Velzimar' et $pos = 0  et $long = 4 la résultat sera: 'Velz'

			
strpos($ch,$motif);

				
Rôle: 
Trouver la première occurence d'un motif dans une chaîne.
(Si elle n'existe pas la Fonction renvoie False)

				
Exemple: 
Pour $ch = 'Velzimar' et $motif = 'Vel' la résultat sera: 0

			
strrpos($ch,$motif);

				
Rôle: 
Trouver la dernière occurence d'un motif dans une chaîne.
(Si elle n'existe pas la Fonction renvoie False)
				
				
Exemple: 
Pour $ch = "Velzimar j'en ai marre!" et $motif = 'mar' la résultat sera: 17

			
strrev($ch);

				
Rôle: 
Inverser l'ordre des caractères d'une chaîne.
				
				
Exemple: 
Pour $ch = "Velzimar" la résultat sera: 'ramizleV'

			
str_replace($ch1,$ch2,$ch);

				
Rôle: 
Remplacer ch1 par ch2 se trouvant dans ch.
				
				
Exemple: 
Pour $ch = "Karkadan" et $ch1 = 'dan' et $ch2 = 'fan' la résultat sera: 'Karkafan'

			
split($Separator,$ch);

				
Rôle: 
Découper une chaîne à base d'un séparateur(Dans un tableau).
				
				
Exemple: 
Pour $ch = "Vel'zimar" et $Separator = "'" la résultat de print $res[0] sera: 'Vel'

Fonctions utiles pour les tableaux

count($var_tableau);

				
Rôle: 
Compter le nombre d'éléments affectés.

				
Exemple: 
Pour $T = array("A","B"); la résultat sera: 2

			
current($var_tableau);

				
Rôle: 
Déterminer la valeur de l'élément en cours.

				
Exemple: 
Pour $T[3] = "C"; la résultat sera: "C"

			
key($var_tableau);

				
Rôle: 
Déterminer l'indice de l'élément en cours.

				
Exemple: 
Pour $T[3] = "C"; la résultat sera: 3

			
list($i,$val); each($var_tableau);

				
Rôle: 
Parcourir les éléments même si les indices ne sont pas consécutifs.

				
Exemple: 

$T[0] = "A";
$T[1] = "B";
$T[2] = "C";
while (list($i,$val) = each($T)){
	echo $i ." - ". $val;
}
la résultat sera:
"0 - A"
"1 - B"
"2 - C"

			
sort($var_tableau);

				
Rôle: 
Trier les éléments et réaffecte les indices.

				
Exemple: 

$T[0] = "50";
$T[1] = "6";
$T[2] = "12";
while (list($i,$val) = each($T)){
	echo $i ." - ". $val;
}
la résultat sera:
"0 - 6"
"1 - 12"
"2 - 50"			 
				

			
asort($var_tableau);

				
Rôle: 
Trier les éléments et ne réaffecter pas les indices.

				
Exemple: 

$T[0] = "50";
$T[1] = "6";
$T[2] = "12";
while (list($i,$val) = each($T)){
	echo $i ." - ". $val;
}
la résultat sera:
"1 - 6"
"2 - 12"
"0 - 50"			 
				

			
ksort($var_tableau);

				
Rôle: 
Trier les éléments selon leur indice.
	 
				

			
rsort($var_tableau);

				
Rôle: 
Trier les éléments dans l'ordre inverse et réaffecte les indices.

				
Exemple: 

$T[0] = "50";
$T[1] = "6";
$T[2] = "12";
while (list($i,$val) = each($T)){
	echo $i ." - ". $val;
}
la résultat sera:
"0 - 6"
"1 - 12"
"2 - 50"			 
				

			
arsort($var_tableau);

				
Rôle: 
Trier les éléments dans l'ordre inverse et ne réaffecter pas les indices.

				
Exemple: 

$T[0] = "50";
$T[1] = "6";
$T[2] = "12";
while (list($i,$val) = each($T)){
	echo $i ." - ". $val;
}
la résultat sera:
"1 - 6"
"2 - 12"
"0 - 50"			 
				

			
krsort($var_tableau);

				
Rôle: 
Trier les éléments dans l'orde inverse selon leur indice.

Fonctions concernant le type date

checkdate($m,$j,$a);

				
Rôle: 
Retourner true si la date donnée est valide sinon retourner false.

				
Exemple: 
Pour $m = 10 et $j = 5 et $a = 1997 la résultat sera: true

			
date('Y-m-d H:i:s',[$timestamp]);

				
Rôle: 
Retourner une date selon le timestamp ou retourner la date courante si le timestamp 
n'est pas saisit.

				
Exemple: 

$date = date('Y-m-d H:i:s',334926792729792);
la résultat de echo $date sera: "2000-01-16 12:00:16"

			
getdate([$timestamp]);

				
Rôle: 
Retourner un tableau contenant les informations suivantes:
['seconds','minutes','hours','mday','wday','mon','year','yday','weekday','month']

				
Exemple: 

$a = getdate();
echo "le jour du mois: ".$a["mday"]." -- ";
echo "le jour de la semaine: ".$a["wday"]." -- ";
echo "le jour d'année: ".$a["yday"]." -- ";
echo "le nom du jour de la semaine: ".$a["weekday"]." -- ";
echo "le nom du mois: ".$a["month"];
la résultat sera:
le jour du mois: 27 -- le jour de la semaine: 0 -- le jour d'année: 86 -- le nom du jour de la semaine: Sunday -- le nom du mois: March
				

			
time();

				
Rôle: 
Retourner le timestamp courante.

			
mktime($h,$min,$s,$m,$j,$y);

				
Rôle: 
Retourner le timestamp correspondant aux arguments fournis.

				
Exemple: 

$ns = mktime(12,12,12,03,28,2000);				
$date = date('Y-d-m H:i:s',$ns);
echo 'Ce nombre de secondes $ns correspond a cette date $date';
la résultat sera:
Ce nombre de secondes 954241932 correspond a cette date 2000-28-03 12:12:12


			
strtotime($h,$min,$s,$m,$j,$y);

				
Rôle: 
Essayer de lire une date au format anglais et de la transformer en timestamp.

				
Exemple: 

$one = strtotime('10 september 2000');
$plus = strtotime('10 september 2000 + 1day');
echo "
Aujourd'hui est: ". $date = date('Y-d-m H:i:s',$one)." Le jour suivant est: ".$date = date('Y-d-m H:i:s',$plus);
la résultat sera:
Aujourd'hui est: 2000-10-09 00:00:00 Le jour suivant est: 2000-11-09 00:00:00
